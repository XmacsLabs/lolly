95void
96std_ostream_rep::write (const char* s) {
97 if (file && is_w) {
98 if (0 <= fprintf (file, "%s", s)) {
99 const char* c= s;
100 while (*c != 0 && *c != '\n')
101 ++c;
102 if (*c == '\n') flush ();
103 }

56string_to_scheme_tree (string s, int& i) {
57 for (; i < N (s); i++)
58 switch (s[i]) {
59
60 case ' ':
61 case '\t':
62 case '\n':
63 break;
64 case '(': {
65 scheme_tree p (TUPLE);
66 i++;
67 while (true) {
68 while ((i < N (s)) && is_spc (s[i]))
69 i++;
70 if ((i == N (s)) || (s[i] == ')')) break;
71 p << string_to_scheme_tree (s, i);
72 }
73 if (i < N (s)) i++;
74 return p;
75 }
76
77 case '\'':
78 i++;
79 return scheme_tree (TUPLE, "\'", string_to_scheme_tree (s, i));
80
81 case '\"': { // "
82 int start= i++;
83 while ((i < N (s)) && (s[i] != '\"')) { // "
84 if ((i < N (s) - 1) && (s[i] == '\\')) i++;
85 i++;
86 }
87 if (i < N (s)) i++;
88 return scheme_tree (unslash (s (start, i)));
89 }
90
91 case ';':
92 while ((i < N (s)) && (s[i] != '\n'))
93 i++;
94 break;
95
96 default: {
97 int start= i;
98 while ((i < N (s)) && (!is_spc (s[i])) && (s[i] != '(') &&
99 (s[i] != ')')) {
100 if ((i < N (s) - 1) && (s[i] == '\\')) i++;
101 i++;
102 }
103 return scheme_tree (unslash (s (start, i)));
104 }
105 }
106
107 return "";
108}

135string
136slash (string s) {
137 int i, n= N (s);
138 string r;
139 for (i= 0; i < n; i++)
140 switch (s[i]) {
141 case '(':
142 case ')':
143 case ' ':
144 case '\'':
145 if ((n < 2) || (s[0] != '\042') || (s[n - 1] != '\042')) r << "\\";
146 r << s[i];
147 break;
148 case '\\':
149 r << '\\' << s[i];
150 break;
151 case '\042':
152 if (((i == 0) && (s[n - 1] == '\042')) ||
153 ((i == (n - 1)) && (s[0] == '\042')))
154 r << s[i];
155 else r << "\\" << s[i];
156 break;
157 case ((char) 0):
158 r << "\\0";
159 break;
160 case '\t':
161 r << "\\t";
162 break;
163 case '\n':
164 r << "\\n";
165 break;
166 default:
167 r << s[i];
168 }
169 return r;
170}

187scheme_tree_to_string (string& out, scheme_tree p) {
188 if (!is_tuple (p)) {

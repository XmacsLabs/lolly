243is_id (string s) {
244 int i= 0, n= N (s);
245 if (n == 0) return false;
246 for (i= 0; i < n; i++) {
247 if ((i > 0) && (s[i] >= '0') && (s[i] <= '9')) continue;
248 if ((s[i] >= 'a') && (s[i] <= 'z')) continue;
249 if ((s[i] >= 'A') && (s[i] <= 'Z')) continue;
250 if (s[i] == '_') continue;

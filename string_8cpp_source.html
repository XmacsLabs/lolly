241is_id (string s) {
242 int i= 0, n= N (s);
243 if (n == 0) return false;
244 for (i= 0; i < n; i++) {
245 if ((i > 0) && (s[i] >= '0') && (s[i] <= '9')) continue;
246 if ((s[i] >= 'a') && (s[i] <= 'z')) continue;
247 if ((s[i] >= 'A') && (s[i] <= 'Z')) continue;
248 if (s[i] == '_') continue;

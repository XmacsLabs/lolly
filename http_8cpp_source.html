41static tree
42response_to_tree (cpr::Response r, string url) {
43 tree ret= http_response_init ();
44 http_response_set (ret, STATUS_CODE, as<long, tree> (r.status_code));
45 http_response_set (ret, TEXT, tree (r.text.c_str ()));
46 http_response_set (ret, URL, tree (url));
47 http_response_set (ret, ELAPSED, as<double, tree> (r.elapsed));
48 auto hmap= hashmap<string, string> ();
49 for (auto i= r.header.begin (); i != r.header.end (); i++) {
50 string key = locase_all (string (i->first.c_str ()));
51 string value= string (i->second.c_str ());
52 hmap (key) = value;
53 }
54 http_response_set (ret, HEADER, as<hashmap<string, string>, tree> (hmap));
55 return ret;
56}

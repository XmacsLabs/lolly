46static tree
47response_to_tree (cpr::Response r, string url) {
48 tree ret= http_response_init ();
49 http_response_set (ret, STATUS_CODE, as<long, tree> (r.status_code));
50 http_response_set (ret, TEXT, tree (r.text.c_str ()));
51 http_response_set (ret, URL, tree (url));
52 http_response_set (ret, ELAPSED, as<double, tree> (r.elapsed));
53 auto hmap= hashmap<string, string> ();
54 for (auto i= r.header.begin (); i != r.header.end (); i++) {
55 string key = locase_all (string (i->first.c_str ()));
56 string value= string (i->second.c_str ());
57 hmap (key) = value;
58 }
59 http_response_set (ret, HEADER, as<hashmap<string, string>, tree> (hmap));
60 return ret;
61}

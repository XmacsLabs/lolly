48static tree
49response_to_tree (cpr::Response r, string url) {
50 tree ret= http_response_init ();
51 http_response_set (ret, STATUS_CODE, as<long, tree> (r.status_code));
52 http_response_set (ret, TEXT, tree (r.text.c_str ()));
53 http_response_set (ret, URL, tree (url));
54 http_response_set (ret, ELAPSED, as<double, tree> (r.elapsed));
55 auto hmap= hashmap<string, string> ();
56 for (auto i= r.header.begin (); i != r.header.end (); i++) {
57 string key = locase_all (string (i->first.c_str ()));
58 string value= string (i->second.c_str ());
59 hmap (key) = value;
60 }
61 http_response_set (ret, HEADER, as<hashmap<string, string>, tree> (hmap));
62 return ret;
63}

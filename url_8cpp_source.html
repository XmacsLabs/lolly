66string
67url::protocol () {
68 string url_label= this->label ();
69 if (url_label == "root") {
70 return rep->t[1]->label;
71 }
72 else if (url_label == "concat") {
73 return as_url (rep->t[1]).protocol ();
74 }
75 else if (url_label == "or") {
76 // For mutiple urls, the protocol must be consistent
77 // otherwise, it is in empty protocol
78 string p1= as_url (rep->t[1]).protocol ();
79 if (p1 == "") return ""; // if empty protocol, return empty protocol fast
80
81 string p2= as_url (rep->t[2]).protocol ();
82 if (p1 == p2) return p1;
83 else return ""; // if the protocol does not match, return empty protocol
84 }
85 else {
86 return "default";
87 }
88}

114skip_ipv6 (string s, int& i) {
115 i++;
116 while (i < N (s) &&
117 (s[i] == ':' || (s[i] >= '0' && s[i] <= '9') ||
118 (s[i] >= 'a' && s[i] <= 'f') || (s[i] >= 'A' && s[i] <= 'F')))
119 i++;
120 if (i < N (s) && s[i] == ']') i++;
121}

123static bool
124heuristic_is_path (string name, int type) {
125 char sep= (type == 0) ? URL_SEPARATOR : ':';
126 int i= 0, n= N (name);
127 while (i < n)
128 if (name[i] == '[') skip_ipv6 (name, i);
129 else if (name[i] == sep) return true;
130 else i++;

171static url
172url_get_atom (string s, int type) {
173 if (type < URL_STANDARD) {
174 if (s == "~") return url_system (get_env ("HOME"));
175 if (starts (s, "$")) {
176 string val= get_env (s (1, N (s)));
177 if (val == "") return url_none ();
178 return unblank (url_system (val));
179 }
180 }
181 if (occurs ("*", s)) return url_wildcard (s);
182#ifdef WINPATHS
183 if (N (s) == 2 && ends (s, ":"))
184 s->resize (1); // remove the ':' after unit letter
185#endif
186 return as_url (tree (s));

189url
190url_get_name (string s, int type, int i) {
191 char sep = (type == URL_SYSTEM) ? URL_CONCATER : '/';
192 int start= i, n= N (s);
193 while ((i < n) && (s[i] != sep) && (s[i] != '/')) {
194 if (s[i] == '[') skip_ipv6 (s, i);
195 else i++;
196 }
197 url u= url_get_atom (s (start, i), type);
198 // url u= tree (s (start, i));
199 if (i == n) return u;
200 if (start == i) return url_get_name (s, type, i + 1);
201 return u * url_get_name (s, type, i + 1);
202}

204static url
205url_get_path (string s, int type= URL_STANDARD, int i= 0) {
206 char sep = (type == URL_SYSTEM) ? URL_SEPARATOR : ':';
207 int start= i, n= N (s);
208 if (i == n) return url_none ();
209 while ((i < n) && (s[i] != sep)) {
210 if (s[i] == '[') skip_ipv6 (s, i);
211 else i++;
212 }
213 url u= url_general (s (start, i), type);
214 if (i == n) return u;
215 if (start == i) return url_get_path (s, type, i + 1);
216 return u | url_get_path (s, type, i + 1);
217}

245url
246url_general (string name, int type= URL_SYSTEM) {
247 if (starts (name, "local:")) return file_url (name (6, N (name)));
248 if (starts (name, "file://")) return file_url (name (7, N (name)));
249 if (starts (name, "http://")) return http_url (name (7, N (name)));
250 if (starts (name, "https://")) return https_url (name (8, N (name)));
251 if (starts (name, "ftp://")) return ftp_url (name (6, N (name)));
252 if (starts (name, "tmfs://")) return tmfs_url (name (7, N (name)));
253 if (starts (name, "//")) return blank_url (name (2, N (name)));
254 if (heuristic_is_path (name, type)) return url_path (name, type);
255 if (heuristic_is_default (name, type)) return url_default (name, type);
256 if (heuristic_is_mingw_default (name, type))
257 return url_mingw_default (name, type);
258 if (type != URL_CLEAN_UNIX) {
259 if (heuristic_is_http (name)) return http_url (name);
260 if (heuristic_is_ftp (name)) return ftp_url (name);
261 }
262 return url_get_name (name, type);
263}

339string
340suffix (url u) {
341 u= tail (u);
342 if (!is_atomic (u)) return "";
343 string s= as_string (u);
344 int i, n= N (s);
345 for (i= n - 1; i >= 0; i--)
346 if (s[i] == '.') break;
347 if ((i > 0) && (i < n - 1)) {
348 string r= s (i + 1, n);
349 while ((N (r) > 0) && (r[N (r) - 1] == '~' || r[N (r) - 1] == '#'))
350 r= r (0, N (r) - 1);
351 return locase_all (r);
352 }
353 return "";
354}

406delta_sub (url base, url u) {
407 if (is_atomic (base)) return u;
408 if (is_concat (base) && is_concat (u) && (base[1] == u[1])) {
409 if (is_special_root (base[1]) && is_concat (base[2]) && is_concat (u[2]) &&

449expand (url u1, url u2) {
450 if (is_or (u1)) return expand (u1[1], u2) | expand (u1[2], u2);
451 if (is_or (u2)) return expand (u1, u2[1]) | expand (u1, u2[2]);
452 if (is_ancestor (u2)) {

499static url
500factor_sorted (url u) {
501 if (!is_or (u)) return u;
502 url v= factor_sorted (u[2]);
503 if (is_concat (u[1])) {
504 if (is_concat (v) && (u[1][1] == v[1])) return u[1][1] * (u[1][2] | v[2]);

534url
535operator* (url u1, url u2) {
536 // cout << "concat " << u1->t << " * " << u2->t << "\n";
537 if (is_root (u2) || (is_concat (u2) && is_root (u2[1]))) {
538 if (is_concat (u1) && is_root_web (u1[1])) {
539 if (is_root (u2, "default") ||
540 (is_concat (u2) && is_root (u2[1], "default"))) {
541 url v= u1[2];
542 while (is_concat (v))
543 v= v[1];
544 if (is_root (u2)) return u1[1] * v;
545 return u1[1] * v * u2[2];
546 }
547 if (is_root (u2, "blank") || (is_concat (u2) && is_root (u2[1], "blank")))
548 return reroot (u2, u1[1][1]->t->label);
549 }
550 return u2;
551 }
552 if (is_here (u1) || (u1->t == "")) return u2;
553 if (is_here (u2)) return u1;
554 if (is_none (u1)) return url_none ();
555 if (is_none (u2)) return url_none ();
556 if (u2 == url_parent ()) {
557 if (is_root (u1)) return u1;
558 if (is_pseudo_atomic (u1) && (!is_parent (u1))) return url_here ();
559 if (is_semi_root (u1)) return u1;
560 }
561 if (is_concat (u2) && (u2[1] == url_parent ())) {
562 if (is_root (u1)) return u1 * u2[2];
563 if (is_pseudo_atomic (u1) && (!is_parent (u1))) return u2[2];
564 if (is_semi_root (u1)) return u1 * u2[2];
565 }
566 if (is_concat (u1)) return u1[1] * (u1[2] * u2);
567 return as_url (url_tuple ("concat", u1->t, u2->t));
568}

665string
666as_string (url u, int type) {
667 // This routine pritty prints an url as a string.
668 // FIXME: the current algorithm is quadratic in time.
669 if (is_none (u)) return "{}";
670 if (is_atomic (u)) return u->t->label;
671 if (is_concat (u)) {
672 int stype= type;
673 if (is_root (u[1]) && (!is_root (u[1], "default"))) stype= URL_STANDARD;

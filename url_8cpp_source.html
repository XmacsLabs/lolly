65string
66url::protocol () {
67 string url_label= this->label ();
68 if (url_label == "root") {
69 return rep->t[1]->label;
70 }
71 else if (url_label == "concat") {
72 return as_url (rep->t[1]).protocol ();
73 }
74 else if (url_label == "or") {
75 // For mutiple urls, the protocol must be consistent
76 // otherwise, it is in empty protocol
77 string p1= as_url (rep->t[1]).protocol ();
78 if (p1 == "") return ""; // if empty protocol, return empty protocol fast
79
80 string p2= as_url (rep->t[2]).protocol ();
81 if (p1 == p2) return p1;
82 else return ""; // if the protocol does not match, return empty protocol
83 }
84 else {
85 return "default";
86 }
87}

113skip_ipv6 (string s, int& i) {
114 i++;
115 while (i < N (s) &&
116 (s[i] == ':' || (s[i] >= '0' && s[i] <= '9') ||
117 (s[i] >= 'a' && s[i] <= 'f') || (s[i] >= 'A' && s[i] <= 'F')))
118 i++;
119 if (i < N (s) && s[i] == ']') i++;
120}

170static url
171url_get_atom (string s, int type) {
172 if (type < URL_STANDARD) {
173 if (s == "~") return url_system (get_env ("HOME"));
174 if (starts (s, "$")) {
175 string val= get_env (s (1, N (s)));
176 if (val == "") return url_none ();

179 }
180 if (contains (s, '*')) return url_wildcard (s);
181#ifdef WINPATHS

183 s->resize (1); // remove the ':' after unit letter
184#endif
185 return as_url (tree (s));

188url
189url_get_name (string s, int type, int i) {
190 char sep = (type == URL_SYSTEM) ? URL_CONCATER : '/';
191 int n = N (s);
192 list<url> u_list= list<url> ();
193 while (i < n) {
194 int start= i;
195 while ((i < n) && (s[i] != sep) && (s[i] != '/')) {
196 if (s[i] == '[') skip_ipv6 (s, i);
197 else i++;
198 }
199 if (i > start) {
200 url u = url_get_atom (s (start, i), type);
201 u_list= list (u, u_list);
202 }
203 else if (i == start) {
204 i++;
205 if (i == n) {
206 u_list= list (as_url (tree ("")), u_list);
207 }
208 }
209 }
210
211 if (is_nil (u_list)) return as_url (tree (""));
212 url ret= u_list->item;
213 u_list = u_list->next;
214 while (!is_nil (u_list)) {
215 ret = u_list->item * ret;
216 u_list= u_list->next;
217 }
218 return ret;
219}

221static url
222url_get_path (string s, int type= URL_STANDARD, int i= 0) {
223 char sep = (type == URL_SYSTEM) ? URL_SEPARATOR : ':';
224 int start= i, n= N (s);
225 if (i == n) return url_none ();
226 while ((i < n) && (s[i] != sep)) {
227 if (s[i] == '[') skip_ipv6 (s, i);
228 else i++;
229 }
230 url u= url_general (s (start, i), type);
231 if (i == n) return u;
232 if (start == i) return url_get_path (s, type, i + 1);
233 return u | url_get_path (s, type, i + 1);
234}

262url
263url_general (string name, int type= URL_SYSTEM) {
264 if (contains (name, "://")) {
265 array<string> tokens = tokenize (name, "://");
266 string protocol= tokens[0];
267 string path = tokens[1];
268 if (N (tokens) == 2 && is_alphanum (protocol) && N (protocol) >= 2) {
269 if (protocol == "file") return file_url (path);
270 if (protocol == "http") return http_url (path);
271 if (protocol == "https") return https_url (path);
272 if (protocol == "ftp") return ftp_url (path);
273 return url_root (tokens[0]) * url_get_name (tokens[1], type);
274 }
275 }
276 if (starts (name, "local:")) {
277 return file_url (name (6, N (name)));
278 }
279 if (starts (name, "//")) {
280 return blank_url (name (2, N (name)));
281 }
282 if (heuristic_is_path (name, type)) {
283 return url_path (name, type);
284 }
285 if (heuristic_is_default (name, type)) return url_default (name, type);
286 if (os_mingw () && heuristic_is_mingw_default (name, type))
287 return url_mingw_default (name, type);
288 if (type != URL_CLEAN_UNIX) {
289 if (heuristic_is_http (name)) return http_url (name);
290 if (heuristic_is_ftp (name)) return ftp_url (name);
291 }
292 return url_get_name (name, type);
293}

369string
370suffix (url u, bool use_locase) {
371 u= tail (u);
372 if (!is_atomic (u)) return "";
373 string s= as_string (u);
374 int i, n= N (s);
375 for (i= n - 1; i >= 0; i--)
376 if (s[i] == '.') break;
377 if ((i > 0) && (i < n - 1)) {
378 string r= s (i + 1, n);
379 while ((N (r) > 0) && (r[N (r) - 1] == '~' || r[N (r) - 1] == '#'))
380 r= r (0, N (r) - 1);
381 int found= index_of (r, '?');
382 if (found != -1) r= r (0, found);
383 if (use_locase) return locase_all (r);
384 else return r;
385 }
386 return "";
387}

487expand (url u1, url u2) {

561bool
562descends (url u, url base) {
563 if (u == base) return true;
564 if (is_concat (u) && is_atomic (base)) return u[1] == base;
565 if (is_concat (u) && is_concat (base)) {
566 url iter_u= u, iter_base= base;
567 while (iter_u[1] == iter_base[1]) {
568 iter_u = iter_u[2];
569 iter_base= iter_base[2];
570 if (is_concat (iter_u) && is_concat (iter_base)) {
571 continue;
572 }
573 else {
574 return descends (iter_u, iter_base);
575 }
576 }
577 return false;
578 }
579 if (is_or (u)) return descends (u[1], base) && descends (u[2], base);
580 if (is_or (base)) return descends (u, base[1]) || descends (u, base[2]);
581 return false;
582}

584url
585operator* (url u1, url u2) {
586 if (is_here (u1) || (u1->t == "")) return u2;
587 if (is_atomic (u1) && is_concat (u2)) {
588 return as_url (url_tuple ("concat", u1->t, u2->t));
589 }
590 if (is_root (u2) || (is_concat (u2) && is_root (u2[1]))) {
591 if (is_concat (u1) && is_root_web (u1[1])) {
592 if (is_root (u2, "default") ||
593 (is_concat (u2) && is_root (u2[1], "default"))) {
594 url v= u1[2];
595 while (is_concat (v))
596 v= v[1];
597 if (is_root (u2)) return u1[1] * v;
598 return u1[1] * v * u2[2];
599 }
600 if (is_root (u2, "blank") || (is_concat (u2) && is_root (u2[1], "blank")))
601 return reroot (u2, u1[1][1]->t->label);
602 }
603 if (is_root (u1) && is_rooted (u1, "file")) {
604 return u1 * u2[2];
605 }
606 return u2;
607 }
608 if (is_here (u2)) return u1;
609 if (is_none (u1)) return url_none ();
610 if (is_none (u2)) return url_none ();
611 if (u2 == url_parent ()) {
612 if (is_root (u1)) return u1;
613 if (is_pseudo_atomic (u1) && (!is_parent (u1))) return url_here ();
614 if (is_semi_root (u1)) return u1;
615 }
616 if (is_concat (u2) && (u2[1] == url_parent ())) {
617 if (is_root (u1)) return u1 * u2[2];
618 if (is_pseudo_atomic (u1) && (!is_parent (u1))) return u2[2];
619 if (is_semi_root (u1)) return u1 * u2[2];
620 }
621 if (is_concat (u1)) return u1[1] * (u1[2] * u2);
622 return as_url (url_tuple ("concat", u1->t, u2->t));
623}

720string
721as_string (url u, int type) {
722 // This routine pretty prints an url as a string.
723 // FIXME: the current algorithm is quadratic in time.
724 if (is_none (u)) return "{}";
725 if (is_atomic (u)) return u->t->label;
726 if (is_concat (u)) {
727 int stype= type;
728 if (is_root (u[1]) &&
729 !(is_root (u[1], "default") || is_root (u[1], "file"))) {
730 stype= URL_STANDARD;
731 }
732 string sep= (stype == URL_SYSTEM ? string (URL_CONCATER) : string ("/"));
733 string s1 = as_string (u[1], type);
734 string s2 = as_string (u[2], stype);
735 if (is_root (u[1], "default")) s1= "";
736 if ((!is_name (u[1])) && (!is_root (u[1]))) s1= "{" * s1 * "}";
737 if ((!is_concat (u[2])) && (!is_atomic (u[2])) && (!is_wildcard (u[2], 1)))
738 s2= "{" * s2 * "}";
739 if (os_win () && // have to return the windows format
740 ((is_root (u[1], "default") && type == URL_SYSTEM) ||
741 is_root (u[1], "file"))) {
742 string root, remain;
743 if (is_concat (u[2])) {
744 root= as_string (u[2][1], type);
745 // root might be unit letter or hostname. It depends on the length
746 remain= as_string (u[2][2], type);
747 }
748 else {
749 root = s2;
750 remain= "";
751 }
752 if (is_root (u[1], "default")) {
753 if (N (root) == 1) return root * ":" * sep * remain; // drive letter
754 else return "\\\\" * root * "\\" * remain;
755 }
756 else {
757 if (N (root) == 1)
758 return s1 * "/" * root * ":" * sep * remain; // local file
759 else return s1 * root * "/" * remain; // remote
760 }
761 }
762 return s1 * sep * s2;
763 }
764 if (is_or (u)) {
765 string s1= as_string (u[1], type);
766 string s2= as_string (u[2], type);
767 if (!is_name_in_path (u[1])) s1= "{" * s1 * "}";
768 if ((!is_or (u[2])) && (!is_name_in_path (u[2]))) s2= "{" * s2 * "}";
769 if (os_win ()) {
770 if (type == URL_STANDARD) return s1 * ":" * s2;
771 else return s1 * string (URL_SEPARATOR) * s2;
772 }
773 else {
774 return s1 * string (URL_SEPARATOR) * s2;
775 }
776 }
777 if (os_win ()) {
778 if (is_root (u, "default")) {
779 int stype= type;
780 if (is_root (u[1]) && (!is_root (u[1], "default"))) stype= URL_STANDARD;
781 if (stype == URL_SYSTEM) return "";
782 else return "/";
783 }
784 }
785 else {
786 if (is_root (u, "default")) return "/";
787 }
788 if (is_root (u, "blank")) return "/";
789 if (is_root (u, "file")) return u[1]->t->label * "://";
790 if (is_root (u)) return u[1]->t->label * ":/";
791 if (is_wildcard (u, 0)) return "**";
792 if (is_wildcard (u, 1)) return u->t[1]->label;
793 TM_FAILED ("bad url");
794 return "";
795}

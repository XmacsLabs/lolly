62void
63bench_start (string task) {
64 // start timer for a given type of task
65 if (timing_level[task] == 0) timing_last (task)= (long) texmacs_time ();
66 timing_level (task)++;
67}

69void
70bench_cumul (string task) {
71 // end timer for a given type of task, but don't reset timer
72 timing_level (task)--;
73 if (timing_level[task] == 0) {
74 long ms= ((long) texmacs_time ()) - timing_last (task);
75 timing_nr (task)++;
76 timing_cumul (task)+= ms;
77 timing_last->reset (task);
78 }
79}

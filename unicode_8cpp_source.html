18unicode_get_range (int code) {
19 if (code <= 0x7f) return "ascii";
20 else if (code >= 0x80 && code <= 0x37f) return "latin";
21 else if (code >= 0x380 && code <= 0x3ff) return "greek";
22 else if (code >= 0x400 && code <= 0x4ff) return "cyrillic";
23 else if (code >= 0x2460 && code <= 0x24ff) return "enclosed_alphanumerics";
24 else if (code >= 0x3000 && code <= 0x303f) return "cjk";
25 else if (code >= 0x4e00 && code <= 0x9fcc) return "cjk";
26 else if (code >= 0xff00 && code <= 0xffef) return "cjk";
27 else if (code >= 0x3040 && code <= 0x309F) return "hiragana";
28 else if (code >= 0xac00 && code <= 0xd7af) return "hangul";
29 else if (code >= 0x2000 && code <= 0x23ff) return "mathsymbols";
30 else if (code >= 0x2900 && code <= 0x2e7f) return "mathextra";
31 else if (code >= 0x1d400 && code <= 0x1d7ff) return "mathletters";
32 else return "";
33}

74utf16_to_utf8 (string s_u16) {

76 tb_byte_t* idata= tb_malloc_bytes (isize);
77 tb_long_t osize= (tb_long_t) (isize << 2);
78 tb_byte_t* odata= tb_malloc_bytes ((tb_size_t) osize);
79
80 for (tb_size_t i= 0; i < isize; i++) {
81 idata[i]= (tb_byte_t) s_u16[i];
82 }
83
84 osize= tb_charset_conv_data (TB_CHARSET_TYPE_UTF16, TB_CHARSET_TYPE_UTF8,
85 idata, isize, odata, osize);
86
87 string ret ((int) osize);
88 for (tb_size_t i= 0; i < osize; i++) {
89 ret[i]= (char) odata[i];
90 }
91 if (idata) tb_free (idata);

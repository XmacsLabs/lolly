15static inline constexpr int
16round_length (int n) {
17 n= (n + 3) & (0xfffffffc);
18 if (n < 24) return n;
19 int i= 32;
20 while (n > i)
21 i<<= 1;
22 return i;
23}

64template <class T>
65void
66lolly_string_rep<T>::reserve (int new_n) {
67 int old_size= a_N;
68 if (new_n != 0) {
69 if (old_size == 0) {
70 a_N= round_length (new_n);
71 a = tm_new_array<T> (a_N);
72 }
73 else if (old_size < new_n) {
74 a_N= round_length (new_n);

93bool
94is_of_type (url name, string filter) {
95 if (is_ramdisc (name)) return true;
96
97 if (!is_single_path (name)) return false;
98
99 if (filter == "") return true;
100 int i, n= N (filter);
101
102 // Normal files
103#if defined(OS_MINGW) || defined(OS_WIN)
104 string suf;
105 if (filter == "x") {
106 suf= suffix (name);
107 if ((suf != "exe") && (suf != "bat") && (suf != "com")) {
108 name= glue (name, ".exe");
109 suf = "exe";
110 }
111 }
112#endif
113 string path= as_string (name);
114 tb_file_info_t info;
115 if (!tb_file_info (as_charp (path), &info)) {
116 return false;
117 }
118 for (i= 0; i < n; i++)
119 switch (filter[i]) {
120 // FIXME: should check user id and group id for r, w and x
121 case 'f':
122 if (info.type != TB_FILE_TYPE_FILE) return false;
123 break;
124 case 'd':
125 if (info.type != TB_FILE_TYPE_DIRECTORY) return false;
126 break;
127 case 'l':
128 if (info.flags != TB_FILE_FLAG_LINK) return false;
129 break;
130 case 'r':
131 if (!tb_file_access (as_charp (path), TB_FILE_MODE_RO)) return false;
132 break;
133 case 'w':
134 if (!tb_file_access (as_charp (path), TB_FILE_MODE_WO)) return false;
135 break;
136 case 'x':
137 if (!tb_file_access (as_charp (path), TB_FILE_MODE_EXEC)) return false;
138 break;
139 }
140 return true;
141}

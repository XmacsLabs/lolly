24inline bool
25is_alpha (char c) {
26 return ((c >= 'a') && (c <= 'z')) || ((c >= 'A') && (c <= 'Z'));
27}
28inline bool
29is_locase (char c) {
30 return (c >= 'a') && (c <= 'z');
31}
32inline bool
33is_upcase (char c) {
34 return (c >= 'A') && (c <= 'Z');
35}
36inline bool
37is_digit (char c) {
38 return (c >= '0') && (c <= '9');
39}
40inline bool
41is_binary_digit (char c) {
42 return c == '0' || c == '1';
43}
44inline bool
45is_octal_digit (char c) {
46 return c >= '0' && c <= '7';
47}
48inline bool
49is_hex_digit (char c) {
50 return (c >= '0' && c <= '9') || (c >= 'A' && c <= 'F') ||
51 (c >= 'a' && c <= 'f');
52}
53inline bool
54is_numeric (char c) {
55 return ((c >= '0') && (c <= '9')) || (c == '.');
56}
57inline bool
58is_punctuation (char c) {
59 return (c == '.') || (c == ',') || (c == ':') || (c == '\'') || (c == '`') ||
60 (c == ';') || (c == '!') || (c == '?');
61}
62inline bool
63is_space (char c) {
64 return (c == ' ') || (c == '\11') || (c == '\12') || (c == '\15');
65}

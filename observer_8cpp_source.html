37void
38observer_rep::announce (tree& ref, modification mod) {
39 // cout << "Modify: " << mod << "\n";
40 switch (mod->k) {
41 case MOD_ASSIGN:
42 announce_assign (ref, mod->p, mod->t);
43 break;
44 case MOD_INSERT:
45 announce_insert (ref, mod->p, mod->t);
46 break;
47 case MOD_REMOVE:
48 announce_remove (ref, path_up (mod->p), last_item (mod->p));
49 break;
50 case MOD_SPLIT:
51 announce_split (ref, mod->p);
52 break;
53 case MOD_JOIN:
54 announce_join (ref, mod->p);
55 break;
56 case MOD_ASSIGN_NODE:
57 announce_assign_node (ref, mod->p, mod->t->op);
58 break;
59 case MOD_INSERT_NODE:
60 announce_insert_node (ref, mod->p, mod->t);
61 break;
62 case MOD_REMOVE_NODE:
63 announce_remove_node (ref, mod->p);
64 break;
65 case MOD_SET_CURSOR:
66 announce_set_cursor (ref, mod->p, mod->t);
67 break;
68 }
69}

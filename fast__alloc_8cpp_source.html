56void*
57enlarge_malloc (size_t sz) {
58 if (alloc_remains < sz) {
59 if (alloc_remains > 0) {
60 ind (alloc_mem) = alloc_ptr (alloc_remains);
61 alloc_ptr (alloc_remains)= alloc_mem;
62 }
63 alloc_mem= (char*) safe_malloc (BLOCK_SIZE);
64#ifdef DEBUG_ON
65 alloc_mem_top= alloc_mem_top >= alloc_mem + BLOCK_SIZE
66 ? alloc_mem_top
67 : (alloc_mem + BLOCK_SIZE);
68 alloc_mem_bottom=
69 alloc_mem_bottom > alloc_mem ? alloc_mem : alloc_mem_bottom;
70#endif
71 alloc_remains= BLOCK_SIZE;
72 fast_chunks++;
73 }
74 void* ptr= alloc_mem;
75 alloc_mem+= sz;
76 alloc_remains-= sz;
77 return ptr;
78}

80void*
81fast_alloc (size_t sz) {
82 sz= (sz + WORD_LENGTH_INC) & WORD_MASK;
83 if (sz < MAX_FAST) {
84 void* ptr= alloc_ptr (sz);
85 if (ptr == NULL) return enlarge_malloc (sz);
86 alloc_ptr (sz)= ind (ptr);
87#ifdef DEBUG_ON
88 break_stub (ptr);
89#endif
90 return ptr;
91 }
92 else {
93 if (MEM_DEBUG >= 3) cout << "Big alloc of " << sz << " bytes\n";
94 if (MEM_DEBUG >= 3) cout << "Memory used: " << mem_used () << " bytes\n";
95 large_uses+= sz;
96 return safe_malloc (sz);
97 }
98}

119void*
120fast_realloc (void* ptr, size_t old_size, size_t new_size) {
121 size_t old_size_plus_pointer= (old_size + WORD_LENGTH_INC) & WORD_MASK;
122 size_t new_size_plus_pointer= (new_size + WORD_LENGTH_INC) & WORD_MASK;
123 if (old_size_plus_pointer >= MAX_FAST && new_size_plus_pointer >= MAX_FAST) {
124 if (MEM_DEBUG >= 3)
125 cout << "Big realloc from " << old_size << " to " << new_size
126 << " bytes\n";
127 large_uses+= new_size - old_size;
128 if (MEM_DEBUG >= 3) cout << "Memory used: " << mem_used () << " bytes\n";
129 return realloc (ptr, new_size);
130 }
131 else {
132 void* newptr= fast_alloc (new_size);
133 memcpy (newptr, ptr, (old_size < new_size ? old_size : new_size));
134 fast_free (ptr, old_size);
135 return newptr;
136 }

139void*
140fast_new (size_t s) {
141 void* ptr;
142#ifdef DEBUG_ON
143 s= (s + (4 * WORD_LENGTH) + WORD_LENGTH_INC) & WORD_MASK;
144#else
145 s = (s + WORD_LENGTH + WORD_LENGTH_INC) & WORD_MASK;
146#endif
147 if (s < MAX_FAST) {
148 ptr= alloc_ptr (s);
149 if (ptr == NULL) ptr= enlarge_malloc (s);
150 else alloc_ptr (s)= ind (ptr);
151#ifdef DEBUG_ON
152 break_stub (ptr);
153#endif
154 }
155 else {
156 if (MEM_DEBUG >= 3) cout << "Big alloc of " << s << " bytes\n";
157 if (MEM_DEBUG >= 3) cout << "Memory used: " << mem_used () << " bytes\n";
158 ptr= safe_malloc (s);
159 // if ((((int) ptr) & 15) != 0) cout << "Unaligned new " << ptr << "\n";
160 large_uses+= s;
161 }
162#ifdef DEBUG_ON
163 char* mem = (char*) ptr;
164 *((size_t*) ptr) = s;
165 ptr = ((char*) ptr) + WORD_LENGTH;
166 *((size_t*) ptr) = s;
167 ptr = ((char*) ptr) + WORD_LENGTH;
168 *((size_t*) ptr) = ~s;
169 ptr = ((char*) ptr) + WORD_LENGTH;
170 *((int*) (mem + s - WORD_LENGTH))= 0x55aa;
171 return (void*) ptr;
172#else
173 *((size_t*) ptr)= s;
174 return (void*) (((char*) ptr) + WORD_LENGTH);
175#endif

180alloc_check (const char* msg, void* ptr, size_t* sp) {
181 void* mem = ptr;
182 ptr = (void*) (((char*) ptr) - WORD_LENGTH);
183 size_t comp= *((size_t*) ptr);
184 ptr = (void*) (((char*) ptr) - WORD_LENGTH);

191 if (break_stub (ptr)) s= s1 < s ? s1 : s;
192 } // else printf("fast_delete %p size %lu at %p\n",mem,s,ptr);
193 if (*((int*) ((char*) ptr + s - WORD_LENGTH)) != 0x55AA) {
194 printf ("%s buffer overflow %x\n", msg, *((int*) ((char*) ptr + s - 1)));
195 }
196 if (sp) *sp= s;

265int
266mem_used () {
267 int free_bytes= alloc_remains;
268 int chunks_use= BLOCK_SIZE * fast_chunks;
269 int i;
270 for (i= WORD_LENGTH; i < MAX_FAST; i+= WORD_LENGTH)
271 free_bytes+= i * compute_free (alloc_table + i);
272 int small_uses= chunks_use - free_bytes;
273 return small_uses + large_uses;
274}

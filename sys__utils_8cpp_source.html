22string
23get_env (string var) {
24 tb_size_t size = 0;
25 string ret = string ("");
26 tb_environment_ref_t environment= tb_environment_init ();
27 if (environment) {
28 size= tb_environment_load (environment, as_charp (var));
29 if (size >= 1) {
30 tb_for_all_if (tb_char_t const*, value, environment, value) {
31 ret= ret * string (value) * ";";
32 }
33 }
34 }
35 tb_environment_exit (environment);
36
37 if (size <= 0) {
38 return ret;
39 }
40 else {
41 return ret (0, N (ret) - 1);
42 }
43}

45void
46set_env (string var, string with) {
47 if (is_empty (with)) {
48 return;
49 }
50 tb_environment_ref_t environment= tb_environment_init ();
51 if (environment) {
52 tb_environment_insert (environment, as_charp (with), tb_true);
53 tb_environment_save (environment, as_charp (var));
54 tb_environment_exit (environment);
55 }
56}

109array<string>
110evaluate_system (array<string> arg, array<int> fd_in, array<string> in,
111 array<int> fd_out) {
112 array<string> out (N (fd_out));
113 array<string*> ptr (N (fd_out));
114 for (int i= 0; i < N (fd_out); i++)
115 ptr[i]= &(out[i]);
116
117#if defined(OS_MINGW)
118 int ret= win_system (arg, fd_in, in, fd_out, ptr);
119#elif defined(OS_WIN)
120 int ret = -1;
121#else
122 int ret= unix_system (arg, fd_in, in, fd_out, ptr);
123#endif
124
125 return append (as_string (ret), out);
126}

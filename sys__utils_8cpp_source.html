22string
23get_env (string var) {
24 tb_size_t size = 0;
25 string ret = string ("");
26 tb_environment_ref_t environment= tb_environment_init ();
27 if (environment) {
28 size= tb_environment_load (environment, as_charp (var));
29 if (size >= 1) {
30 tb_for_all_if (tb_char_t const*, value, environment, value) {
31 ret= ret * string (value) * URL_SEPARATOR;
32 }
33 }
34 }
35 tb_environment_exit (environment);
36
37 if (size <= 0) {
38 return ret;
39 }
40 else {
41 return ret (0, N (ret) - 1);
42 }
43}

45void
46set_env (string var, string with) {
47 if (is_empty (with)) {
48 return;
49 }
50 tb_environment_ref_t environment= tb_environment_init ();
51 if (environment) {
52 tb_environment_insert (environment, as_charp (with), tb_true);
53 tb_environment_save (environment, as_charp (var));
54 tb_environment_exit (environment);
55 }

120array<string>
121evaluate_system (array<string> arg, array<int> fd_in, array<string> in,
122 array<int> fd_out) {
123 array<string> out (N (fd_out));
124 array<string*> ptr (N (fd_out));
125 for (int i= 0; i < N (fd_out); i++)
126 ptr[i]= &(out[i]);
127
128#if defined(OS_MINGW)
129 int ret= win_system (arg, fd_in, in, fd_out, ptr);
130#elif defined(OS_WIN)
131 int ret = -1;
132#else
133 int ret= unix_system (arg, fd_in, in, fd_out, ptr);
134#endif
135
136 return append (as_string (ret), out);
137}

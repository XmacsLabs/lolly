24string
25get_env (string var) {
26 c_string var_ (var);
27 tb_size_t size = 0;
28 string ret = string ("");
29 tb_environment_ref_t environment= tb_environment_init ();
30 if (environment) {
31 size= tb_environment_load (environment, var_);
32 if (size >= 1) {
33 tb_for_all_if (tb_char_t const*, value, environment, value) {
34 ret= ret * string (value) * URL_SEPARATOR;
35 }
36 }
37 }
38 tb_environment_exit (environment);
39
40 if (size <= 0) {
41 return ret;
42 }
43 else {
44 return ret (0, N (ret) - 1);
45 }
46}

144array<string>
145evaluate_system (array<string> arg, array<int> fd_in, array<string> in,
146 array<int> fd_out) {
147 array<string> out (N (fd_out));
148 array<string*> ptr (N (fd_out));
149 for (int i= 0; i < N (fd_out); i++)
150 ptr[i]= &(out[i]);
151
152#if defined(OS_MINGW)
153 int ret= win_system (arg, fd_in, in, fd_out, ptr);
154#endif
155
156#if defined(OS_LINUX) || defined(OS_MACOS)
157 int ret= unix_system (arg, fd_in, in, fd_out, ptr);
158#endif
159
160#if defined(OS_WIN) || defined(OS_WASM)
161 int ret= -1;
162#endif
163
164 return append (as_string (ret), out);

174string
175get_stacktrace (unsigned int max_frames) {
176 string r;
177 r << "Backtrace of C++ stack:\n";
178
179 // storage array for stack trace address data. skip the first, it is the
180 // address of this function.
181 STACK_NEW_ARRAY (addrlist, tb_pointer_t, max_frames + 1);
182
183 // retrieve current stack addresses
184 int addrlen= tb_backtrace_frames (addrlist, max_frames, 2);
185
186 if (addrlen == 0) {
187 r << " <empty, possibly corrupt>\n";
188 STACK_DELETE_ARRAY (addrlist);
189 return r;
190 }
191
192 // resolve addresses into strings by tbox, differs under various platforms.
193 // this array must be released
194 tb_handle_t symbollist= tb_backtrace_symbols_init (addrlist, addrlen);
195
196 // allocate string which will be filled with the function name
197 char* funcname= tm_new_array<char> (1024);
198
199 // iterate over the returned symbol lines.
200 for (int i= 0; i < addrlen; i++) {
201 // print name of current stack frame
202 const char* curname=
203 tb_backtrace_symbols_name (symbollist, addrlist, addrlen, i);
204 if (curname == NULL) {
205 r << " null\n";
206 }
207 else {
208 r << " " << string (curname) << "\n";
209 }
210 }
211
212 tm_delete_array (funcname);
213 tb_backtrace_symbols_exit (symbollist);
214 STACK_DELETE_ARRAY (addrlist);
215 return r;
216}

138template <typename T> class lolly_tree_rep : concrete_struct {
139public:
140 // the tree_label of op=0 in TeXmacs is STRING
141 int op;
142 T data;
143 union {
144 string label;
145 array<lolly_tree<T>> a;
146 };
147
148 inline lolly_tree_rep (string l) : op (0), label (l) {}
149 inline lolly_tree_rep (int l, array<lolly_tree<T>> a2) : op (l), a (a2) {}
150 inline ~lolly_tree_rep () {
151 if (op == 0) {
152 label.~string ();

154 else if (op > 0) {
155 a.~array ();
156 };
157 }

136template <typename T> class lolly_tree_rep : concrete_struct {
137public:
138 // the tree_label of op=0 in TeXmacs is STRING
139 int op;
140 T data;
141 union {
142 string label;

144 };
145
146 inline lolly_tree_rep (string l) : op (0), label (l) {}
147 inline lolly_tree_rep (int l, array<lolly_tree<T>> a2) : op (l), a (a2) {}
148 inline ~lolly_tree_rep () {
149 if (op == 0) {
150 label.~string ();

152 else if (op > 0) {
153 a.~array ();
154 };

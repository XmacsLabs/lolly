30int
31check_output (string s, string& result, int64_t timeout) {
32 tb_long_t status= -1;
33 // init pipe files
34 tb_pipe_file_ref_t file[2]= {0};
35 if (!tb_pipe_file_init_pair (file, tb_null, 0)) {
36 return status;
37 }
38
39 // init process
40 c_string cmd_ (s);
41 tb_process_attr_t attr = {0};
42 attr.out.pipe = file[1];
43 attr.outtype = TB_PROCESS_REDIRECT_TYPE_PIPE;
44 attr.flags = TB_PROCESS_FLAG_NO_WINDOW;
45 tb_process_ref_t process= tb_process_init_cmd (cmd_, &attr);
46 if (process) {
47 // read pipe data
48 tb_size_t read= 0;
49 // TODO: should be a config here
50 tb_byte_t data[8192];
51 tb_size_t size= sizeof (data);
52 tb_bool_t wait= tb_false;
53 while (read < size) {
54 tb_long_t real= tb_pipe_file_read (file[0], data + read, size - read);
55 if (real > 0) {
56 read+= real;
57 wait= tb_false;
58 }
59 else if (!real && !wait) {
60 tb_long_t ok = 0;
61 int retry= 25;
62 if (read > 0) {
63 retry= 2;
64 }
65 while (retry > 0 && (ok == 0)) {
66 ok = tb_pipe_file_wait (file[0], TB_PIPE_EVENT_READ, timeout);
67 retry= retry - 1;
68 }
69 tb_check_break (ok > 0);
70 wait= tb_true;
71 }
72 else break;
73 }
74
75 result= as_string ((tb_char_t*) data);
76
77 // wait process
78 tb_process_wait (process, &status, timeout);
79
80 // exit process
81 tb_process_exit (process);
82 }
83
84 // exit pipe files
85 tb_pipe_file_exit (file[0]);
86 tb_pipe_file_exit (file[1]);
87 return status;
88}

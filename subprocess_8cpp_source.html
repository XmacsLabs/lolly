30int
31check_output (string s, string& result) {
32 tb_long_t status= -1;
33 // init pipe files
34 tb_pipe_file_ref_t file[2]= {0};
35 if (!tb_pipe_file_init_pair (file, tb_null, 0)) {
36 return status;
37 }
38
39 // init process
40 c_string cmd_ (s);
41 tb_process_attr_t attr = {0};
42 attr.out.pipe = file[1];
43 attr.outtype = TB_PROCESS_REDIRECT_TYPE_PIPE;
44 attr.flags = TB_PROCESS_FLAG_NO_WINDOW;
45 tb_process_ref_t process= tb_process_init_cmd (cmd_, &attr);
46 if (process) {
47 // read pipe data
48 tb_size_t read= 0;
49 // TODO: should be a config here
50 tb_byte_t data[8192];
51 tb_size_t size= sizeof (data);
52 tb_bool_t wait= tb_false;
53 while (read < size) {
54 tb_long_t real= tb_pipe_file_read (file[0], data + read, size - read);
55 if (real > 0) {
56 read+= real;
57 wait= tb_false;
58 }
59 else if (!real && !wait) {
60 // wait pipe
61 tb_long_t ok= tb_pipe_file_wait (file[0], TB_PIPE_EVENT_READ, 1000);
62 tb_check_break (ok > 0);
63 wait= tb_true;
64 }
65 else break;
66 }
67
68 result= as_string ((tb_char_t*) data);
69
70 // wait process
71 tb_process_wait (process, &status, -1);
72
73 // exit process
74 tb_process_exit (process);
75 }
76
77 // exit pipe files
78 tb_pipe_file_exit (file[0]);
79 tb_pipe_file_exit (file[1]);
80 return status;
81}

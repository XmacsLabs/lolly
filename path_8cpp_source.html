116bool
117path_less_eq (path p1, path p2) {
118 if (is_nil (p1) || is_nil (p2)) return p1 == p2;
119 if (is_atom (p1) || is_atom (p2)) {
120 if (is_atom (p1) && is_atom (p2)) return p1->item <= p2->item;
121 if ((p1->item == 0) && is_nil (p1->next)) return true;
122 if ((p2->item == 1) && is_nil (p2->next)) return true;
123 return false;
124 }
125 if (p1->item < p2->item) return true;
126 if (p1->item > p2->item) return false;
127 return path_less_eq (p1->next, p2->next);

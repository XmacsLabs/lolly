141#ifdef OS_MACOS
142string
143get_mac_language () {
144 char mac_lang[50];
145 CFLocaleRef locale= CFLocaleCopyCurrent ();
146 CFTypeRef lang = CFLocaleGetValue (locale, kCFLocaleLanguageCode);
147 CFStringGetCString ((CFStringRef) lang, mac_lang, sizeof (mac_lang),
148 kCFStringEncodingUTF8);
149 CFRelease (locale);
150 return string (mac_lang);

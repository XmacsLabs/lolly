28int
29win_system (array< ::string> arg, array<int> fd_in, array< ::string> str_in,
30 array<int> fd_out, array< ::string*> str_out) {
31 // Run command arg[0] with arguments arg[i], i >= 1.
32 // str_in[i] is sent to the file descriptor fd_in[i].
33 // str_out[i] is filled from the file descriptor fd_out[i].
34 // If str_in[i] is -1 then $$i automatically replaced by a valid
35 // file descriptor in arg.
36 if (N (arg) == 0) return 0;
37 ::string which= recompose (arg, " ");
38 int n_in= N (fd_in), n_out= N (fd_out);
39 ASSERT (N (str_in) == n_in, "size mismatch");
40 ASSERT (N (str_out) == n_out, "size mismatch");
41 array<Channel> ch (n_in + n_out);
42
43 for (int i= 0; i < n_in; i++) {
44 int fd= fd_in[i];
45 if (fd >= 0) ch[i].Init (fd, Channel::CHIN);
46 else ch[i].Init (Channel::CHIN);
47 }
48 for (int i= 0; i < n_out; i++) {
49 int fd= fd_out[i];
50 if (fd >= 0) ch[i + n_in].Init (fd, Channel::CHOUT);
51 else ch[i + n_in].Init (Channel::CHOUT);
52 }
53
54 array< ::string> arg_= arg;
55 for (int j= 0; j < N (arg); j++)
56 for (int i= 0; i < n_in; i++)
57 if (fd_in[i] < 0) {
58 arg_[j]= replace (arg_[j], "$%" * as_string (i),
59 as_string (_get_osfhandle (ch[i].getPipe ())));
60 arg_[j]= replace (arg_[j], "$$" * as_string (i),
61 as_string (ch[i].getPipe ()));
62 }
63 array<char*> _arg;
64 for (int j= 0; j < N (arg_); j++)
65 _arg << as_charp (arg_[j]);
66 _arg << (char*) NULL;
67
68 spawn_system process (ch, _arg[0], A (_arg));
69 for (int j= 0; j < N (arg_); j++)
70 tm_delete_array (_arg[j]);
71 if (!process.isRunning ()) {
72 return -1;
73 }
74 process.getpid () << LF;
75
76 // receive data from spawn process
77 // class string is not thread safe, use std::string instead
78 array<std::string> str (n_out);
79 for (int i= 0; i < n_out; i++)
80 ch[i + n_in].read (&str[i]);
81
82 // send data to spawn process
83 array<int> pos_in (n_in);
84 for (int i= 0; i < n_in; i++)
85 pos_in[i]= 0;
86 time_t last_wait_time= texmacs_time ();
87
88 bool busy;
89 do {
90 busy= false;
91 if (texmacs_time () - last_wait_time > 5000) { // FIXME?
92 last_wait_time= texmacs_time ();
93 _unix_system_warn (process.getpid (), which, "waiting spawn process");
94 }
95 for (int i= 0; i < n_in; i++) {
96 if (N (str_in[i]) > pos_in[i]) {
97 int m= min (ch[i].sz, N (str_in[i]) - pos_in[i]); // do not fill the
98 // pipe
99 int o= ch[i].write (&(str_in[i][pos_in[i]]), m);
100 if (o >= 0) {
101 pos_in[i]+= o;
102 if (N (str_in[i]) == pos_in[i]) ch[i].close ();
103 else busy= true;
104 }
105 }
106 }
107 } while (busy);
108
109 // wait for process
110 int wret= process.wait ();
111 for (int i= 0; i < n_out; ++i)
112 (*(str_out[i])) << ::string (str[i].data (), str[i].length ());
113 return (wret);
114}

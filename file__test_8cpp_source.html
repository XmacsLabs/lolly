173 tb_byte_t* buffer= (tb_byte_t*) tb_malloc_bytes (size);
174 int seek = 0;
175 while (seek < size) {
176 tb_byte_t c = s2[seek];
177 buffer[seek]= c;
178 seek++;
179 }
180

182 tb_file_writ (file_ref, buffer, strlen (s2));

185 string_eq (s, string ("hello world"));
186 }

73void
74mem_err_handler (int err, void* arg) {
75 switch (err) {
76 case ENOMEM:
77 cerr << "Fatal error: out of memory\n";
78 break;
79 case EINVAL:
80 cerr << "Fatal error: invalid pointer\n";
81 break;
82 case EFAULT:
83 cerr << "Corrupted heap. Please check whether there are objects used after "
84 "free.\n";
85 break;
86 case EOVERFLOW:
87 cerr << "Fatal error: input argument of allocator is too large\n";
88 break;
89 case EAGAIN:
90 cerr << "Double free detected\n";
91 break;
92 default:
93 break;
94 }
95 abort ();
96}

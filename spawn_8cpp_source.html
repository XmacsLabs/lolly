26void
27Channel::Init (Direction d) {
28 int pp[2];
29 if (_pipe (pp, sz, O_NOINHERIT | _O_BINARY) == 0) {
30 int pr[2];
31 if (_pipe (pr, sz, _O_BINARY) == 0) {
32 _close (pr[d]);
33 _dup2 (pp[otherDirection (d)], pr[otherDirection (d)]);
34 _close (pp[otherDirection (d)]);
35 fd = pp[d];
36 toBeClosed= pr[otherDirection (d)];
37 }
38 else _close (pp[d]);

110bkgread (void* thatv) {
111 char buf[1024];
112 int cnt;
113 Channel* that= (Channel*) thatv;
114 do {
115 cnt= _read (that->fd, buf, sizeof (buf));
116 if (cnt > 0) *(that->str)+= std::basic_string<char> (buf, cnt);
117 else if (cnt == 0) that->close ();
